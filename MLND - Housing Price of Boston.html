def performance_metric2(y_true, y_predict):
    """"""
    
    # Type conversion
    y_true = np.array(y_true)
    y_predict = np.array(y_predict)
    
    # The mean of the observed data
    y_mean = np.mean(y_true)
    
    # The total sum of squares
    SS_tot = sum((y_true - y_mean)**2)
    
    # The explained sum of squares
    SS_reg = sum((y_predict - y_mean)**2)
    
    # The residual sum of squares
    SS_res = sum((y_true - y_predict)**2)
    
    score = 1 - SS_res/SS_tot

    return score

from sklearn.tree import DecisionTreeRegressor

def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    
    # 
    n1 = len(X)/3
    n2 = 2*n1
    
    # Xy
    X1 = X[:n1]
    y1 = y[:n1]
    X2 = X[n1:n2]
    y2 = y[n1:n2]
    X3 = X[n2:]
    y3 = y[n2:]
    
    # 
    pred1 = estimator
    pred2 = estimator
    pred3 = estimator
    
    # 
    pred1.fit(np.append(X2, X3, axis=0), np.append(y2, y3, axis=0))
    y1_predict = pred1.predict(X1)
    
    pred2.fit(np.append(X1, X3, axis=0), np.append(y1, y3, axis=0))
    y2_predict = pred2.predict(X2)
    
    pred3.fit(np.append(X1, X2, axis=0), np.append(y1, y2, axis=0))
    y3_predict = pred3.predict(X3)
      
    # 
    scores = [performance_metric(y1,y1_predict),performance_metric(y2,y2_predict),performance_metric(y3,y3_predict)]
    
    return scores
